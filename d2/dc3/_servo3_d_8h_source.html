147  std::shared_ptr<SceneBase> make_scene() override {
148  std::shared_ptr<SceneBase> scene = std::make_shared<SceneBase>(SceneBase());
149 
150  /* Materials */
151  std::shared_ptr<Material> particleMat = std::make_shared<FrictMat>(
152  FrictMat(stiffness, stiffness, friction, density, damping, damping));
153  std::shared_ptr<Material> wallMat = std::make_shared<FrictMat>(FrictMat(
154  wallStiffness, wallStiffness, wallFriction, density, wallDamping, wallDamping));
155  scene->addMaterial(particleMat);
156  scene->addMaterial(wallMat);
157  /* Bodies */
158  /* Bodies */
159  double xItv = 0, yItv = 0, zItv = 0;
160  if (xNum > 1)
161  xItv = (xRight - xLeft - 4 * radius) / (xNum - 1);
162  if (yNum > 1)
163  yItv = (yUp - yBottom - 4 * radius) / (yNum - 1);
164  if (zNum > 1)
165  zItv = (zFront - zBack - 4 * radius) / (zNum - 1);
166  for (int i = 0; i < xNum; i++) {
167  double x = (xLeft + xRight) / 2.0 + (i + 0.5 - 0.5 * xNum) * xItv;
168  for (int j = 0; j < yNum; j++) {
169  double y = (yBottom + yUp) / 2.0 + (j + 0.5 - 0.5 * yNum) * yItv;
170  for (int k = 0; k < zNum; k++) {
171  double z = (zBack + zFront) / 2.0 + (k + 0.5 - 0.5 * zNum) * zItv;
172  double x1 = x + (radius * randDouble(-0.25, 0.25));
173  double z1 = z + (radius * randDouble(-0.25, 0.25));
174  double rd = radius * randDouble(1 - radiusRnd, 1 + radiusRnd);
175  scene->addBody(std::make_shared<SphereShp>(SphereShp(rd)), particleMat,
176  std::make_shared<NormalState>(NormalState(vec3(x1, y, z1))),
177  std::make_shared<AabbBnd>(AabbBnd()));
178  }
179  }
180  }
181 
182  scene->addWalls(std::make_shared<RectangleShp>(
183  RectangleShp(RectAlignment::XZ, xRight - xLeft, zFront - zBack)),
184  wallMat, std::make_shared<NormalState>(NormalState(vec3(0, yUp, 0))),
185  std::make_shared<AabbBnd>(AabbBnd())); // up
186  scene->addWalls(std::make_shared<RectangleShp>(
187  RectangleShp(RectAlignment::XZ, xRight - xLeft, zFront - zBack)),
188  wallMat, std::make_shared<NormalState>(NormalState(vec3(0, yBottom, 0))),
189  std::make_shared<AabbBnd>(AabbBnd())); // bottom
190  scene->addWalls(std::make_shared<RectangleShp>(
191  RectangleShp(RectAlignment::YZ, (yUp - yBottom), (zFront - zBack))),
192  wallMat, std::make_shared<NormalState>(NormalState(vec3(xLeft, 0, 0))),
193  std::make_shared<AabbBnd>(AabbBnd())); // left
194  scene->addWalls(std::make_shared<RectangleShp>(
195  RectangleShp(RectAlignment::YZ, (yUp - yBottom), (zFront - zBack))),
196  wallMat, std::make_shared<NormalState>(NormalState(vec3(xRight, 0, 0))),
197  std::make_shared<AabbBnd>(AabbBnd())); // right
198  scene->addWalls(std::make_shared<RectangleShp>(
199  RectangleShp(RectAlignment::XY, (xRight - xLeft), (yUp - yBottom))),
200  wallMat, std::make_shared<NormalState>(NormalState(vec3(0, 0, zBack))),
201  std::make_shared<AabbBnd>(AabbBnd())); // back
202  scene->addWalls(std::make_shared<RectangleShp>(
203  RectangleShp(RectAlignment::XY, (xRight - xLeft), (yUp - yBottom))),
204  wallMat, std::make_shared<NormalState>(NormalState(vec3(0, 0, zFront))),
205  std::make_shared<AabbBnd>(AabbBnd())); // front
206  /* Functors */
207  scene->addFunctor<SphereShp, SphereShp>(
208  std::make_shared<ig2_Sphere_Sphere_ScGeom>(ig2_Sphere_Sphere_ScGeom()));
209  scene->addFunctor<SphereShp, RectangleShp>(
210  std::make_shared<ig2_Sphere_Rectangle_ScGeom>(ig2_Sphere_Rectangle_ScGeom()));
211  scene->addIPhysFunctor<FrictMat, FrictMat>(
212  std::make_shared<ip2_Frict_Frict_FrictIPhys>(ip2_Frict_Frict_FrictIPhys()));
213  scene->addLawFunctor<FrictPhys, GeneralCircleGeom>(
214  std::make_shared<law2_FrictFrict_GcGeom_Std>(law2_FrictFrict_GcGeom_Std()));
215  scene->addBIntersectFunctor<AabbBnd, AabbBnd>(
216  std::make_shared<bi2_Aabb_Aabb>(bi2_Aabb_Aabb()));
217  /* Engines */
218  if (useGravity) {
219  std::shared_ptr<Engine> gravityEngine =
220  std::make_shared<GravityEngine>(GravityEngine(scene));
221  scene->_gravityEngine = gravityEngine;
222  }
223  std::shared_ptr<Engine> dtCalculator =
224  std::make_shared<CriticalTimestep>(CriticalTimestep(scene, 0.1 * dtRatio));
225  if (useLocalDamping) {
226  std::shared_ptr<Engine> localDamp =
227  std::make_shared<LocalDampingEngine>(LocalDampingEngine(scene, localDamping));
228  scene->_localDamping = localDamp;
229  }
230  scene->_dtCalculator = dtCalculator;
231  std::shared_ptr<Servo3DEngine> servoEngine =
232  std::make_shared<Servo3DEngine>(Servo3DEngine(scene));
233  servoEngine->et3_wsxx_req = et3_wsxx_req;
234  servoEngine->et3_wsyy_req = et3_wsyy_req;
235  servoEngine->et3_wszz_req = et3_wszz_req;
236  servoEngine->et3_servo_xon = et3_servo_xon;
237  servoEngine->et3_servo_yon = et3_servo_yon;
238  servoEngine->et3_servo_zon = et3_servo_zon;
239  servoEngine->tol = tol;
240  servoEngine->_et3_gain_cnt = _et3_gain_cnt;
241  servoEngine->et3_servo_gain_cyc = et3_servo_gain_cyc;
242  servoEngine->et3_servo_alpha = et3_servo_alpha;
243  servoEngine->et3_servo_vmax = et3_servo_vmax;
244  scene->_servoEngine = servoEngine;
245 
246  return scene;
