65  scene->addMaterial(particleMat);

68  /* Bodies */
69  double xItv = 0, yItv = 0, zItv = 0;
70  if (xNum > 1)

72  if (yNum > 1)
73  yItv = (yUp - yBottom - 4 * radius) / (yNum - 1);
74  if (zNum > 1)
75  zItv = (zFront - zBack - 4 * radius) / (zNum - 1);
76  for (int i = 0; i < xNum; i++) {
77  double x = (xLeft + xRight) / 2.0 + (i + 0.5 - 0.5 * xNum) * xItv;
78  for (int j = 0; j < yNum; j++) {
79  double y = (yBottom + yUp) / 2.0 + (j + 0.5 - 0.5 * yNum) * yItv;
80  for (int k = 0; k < zNum; k++) {
81  double z = (zBack + zFront) / 2.0 + (k + 0.5 - 0.5 * zNum) * zItv;
82  double x1 = x + (radius * randDouble(-0.5, 0.5));
83  double z1 = z + (radius * randDouble(-0.5, 0.5));
84  double rd = radius * randDouble(1 - radiusRnd, 1 + radiusRnd);

109  scene->addWalls(std::make_shared<RectangleShp>(

111  wallMat2, std::make_shared<NormalState>(NormalState(vec3(0, 0, zFront))),
112  std::make_shared<AabbBnd>(AabbBnd()), false); // front
113  /* Functors */
114  scene->addFunctor<SphereShp, RectangleShp>(

116  scene->addFunctor<SphereShp, SphereShp>(
117  std::make_shared<ig2_Sphere_Sphere_ScGeom>(ig2_Sphere_Sphere_ScGeom()));
118  scene->addIPhysFunctor<FrictMat, FrictMat>(
119  std::make_shared<ip2_Frict_Frict_FrictIPhys>(ip2_Frict_Frict_FrictIPhys()));
120  scene->addLawFunctor<FrictPhys, GeneralCircleGeom>(
121  std::make_shared<law2_FrictFrict_GcGeom_Std>(law2_FrictFrict_GcGeom_Std()));
122  scene->addBIntersectFunctor<AabbBnd, AabbBnd>(
123  std::make_shared<bi2_Aabb_Aabb>(bi2_Aabb_Aabb()));
124  /* Engines */
125  std::shared_ptr<Engine> gravityEngine =
126  std::make_shared<GravityEngine>(GravityEngine(scene));
127  std::shared_ptr<Engine> dtCalculator =
128  std::make_shared<CriticalTimestep>(CriticalTimestep(scene, 0.1 * dtRatio));
129  if (useLocalDamping) {
130  std::shared_ptr<Engine> localDamp =
131  std::make_shared<LocalDampingEngine>(LocalDampingEngine(scene, localDamping));
132  scene->_localDamping = localDamp;
