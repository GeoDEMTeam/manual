59  std::shared_ptr<SceneBase> make_scene() {

69  // Generate the particles based on radius expansion method
70  /*
71  Step 1: determine the radii list, whose porosity reach the target;
72  Step 2: scale down the radii list such that the generation porosity is greater than 0.7;
73  Step 3: generate unoverlaply and randomly the scaled-down particles;
74  Step 4: expand the particles gradually to reach the desired porosity;
75  */
76  double mean_volum = PI / 3. * (pow(Rmin, 2) + pow(Rmax, 2) + Rmax * Rmin);
77  int num_ball =
78  static_cast<int>(speciman_height * speciman_width * (1. - porosity) / mean_volum);
79  std::cout << "The required number of balls is " << num_ball << std::endl;
80  double actual_porosity = -1;
81  std::vector<double> radii_vector;
82  std::random_device rd; // Will be used to obtain a seed for the random number engine
83  std::mt19937 gen(rd()); // Standard mersenne_twister_engine seeded with rd()
84  std::uniform_real_distribution<> balls_generator(Rmin, Rmax);
85  while (std::abs((actual_porosity - porosity) / porosity) > 0.005) {
86  double total_vol = 0; // area in 2D case
87  radii_vector.clear();
88  for (long ii = 0; ii < num_ball; ii++) {
89  double current_radius = balls_generator(gen);
90  radii_vector.push_back(current_radius);
91  total_vol += PI * pow(current_radius, 2);
92  }
93  actual_porosity = 1. - total_vol / (speciman_height * speciman_width);
94  }
95  // scale down the radii
96  expansion_factor = pow((1 - porosity) / (1 - porosity_scaledown), 0.5);
97  vec3 pos_candidate;
98  for (std::vector<double>::size_type ii = 0; ii != radii_vector.size(); ++ii) {
99  radii_vector[ii] /= expansion_factor;
100  bool pos_ok = false; // flag for successfully generating unoverlopped particle
101  long current_trial = 0;
102  while (!pos_ok) {
103  // std::cout << "Placing "<< ii << std::endl;
104  std::random_device rd; // Will be used to obtain a seed for the random number engine
105  std::uniform_real_distribution<> x_pos_generator(
106  -0.5 * speciman_width + radii_vector[ii],
107  0.5 * speciman_width - radii_vector[ii]);
108  std::uniform_real_distribution<> y_pos_generator(
109  -0.5 * speciman_height + radii_vector[ii],
110  0.5 * speciman_height - radii_vector[ii]);
111  pos_candidate = vec3(x_pos_generator(rd), y_pos_generator(rd), 0);
112  auto& bodyCon = scene->bodyContainer();
113  bool overlap = false;
114  for (int i = 0; i < scene->numBodies(); i++) {
115  // check if overlap
116  const CircleShp& c0 = dynamic_cast<const CircleShp&>(*(bodyCon[i]->shape()));
117  if ((bodyCon[i]->state()->position() - pos_candidate).norm() <
118  c0.radius() + radii_vector[ii]) {
119  overlap = true;
120  current_trial++;
121  break;
122  };
123  }
124  if (!overlap) {
125  pos_ok = true;
126  }
127  }
128  if (pos_ok) {
129  scene->addBody(std::make_shared<CircleShp>(CircleShp(radii_vector[ii])),
130  particleMat,
131  std::make_shared<NormalState>(NormalState(pos_candidate)),
132  std::make_shared<AabbBnd>(AabbBnd()));
133  std::cout << scene->numBodies() << "th particle has been generated." << std::endl;
134  }
135  }
136

140  std::make_shared<NormalState>(NormalState(vec3(0, 0.5 * speciman_height, 0))),
141  std::make_shared<AabbBnd>(AabbBnd())); // top
142

146  std::make_shared<AabbBnd>(AabbBnd())); // bottom
147

152

154  std::make_shared<LineShp>(LineShp(vec3(0, 1, 0), speciman_height * 2.)), wallMat,
155  std::make_shared<NormalState>(NormalState(vec3(0.5 * speciman_width, 0, 0))),
156  std::make_shared<AabbBnd>(AabbBnd())); // right
